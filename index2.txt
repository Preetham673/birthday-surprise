<script>
// ---------- Page 1 Hearts ----------
const heartsCanvas = document.getElementById('heartsCanvas');
const hCtx = heartsCanvas.getContext('2d');
function resizeHeartsCanvas() { heartsCanvas.width = window.innerWidth; heartsCanvas.height = window.innerHeight; }
resizeHeartsCanvas();
window.addEventListener('resize', resizeHeartsCanvas);

let hearts=[], sparkles=[];

function spawnSparkles(count=100){
  for(let i=0;i<count;i++) sparkles.push({x:Math.random()*window.innerWidth,y:Math.random()*window.innerHeight,radius:Math.random()*2+1,alpha:Math.random(),speed:Math.random()*0.5+0.2});
}
function spawnHearts(count=15){
  for(let i=0;i<count;i++) hearts.push({x:Math.random()*window.innerWidth,y:window.innerHeight+Math.random()*100,size:Math.random()*40+30,speed:Math.random()*2+1,angle:Math.random()*Math.PI*2,rotationSpeed:Math.random()*0.05+0.01});
}
function animateHearts(){
  hCtx.clearRect(0,0,heartsCanvas.width,heartsCanvas.height);
  sparkles.forEach(s=>{
    hCtx.beginPath();
    hCtx.arc(s.x,s.y,s.radius,0,Math.PI*2);
    hCtx.fillStyle=`rgba(255,255,255,${s.alpha})`;
    hCtx.fill();
    s.y-=s.speed; s.alpha-=0.005;
    if(s.alpha<=0){ s.x=Math.random()*window.innerWidth; s.y=window.innerHeight; s.alpha=Math.random(); s.speed=Math.random()*0.5+0.2; }
  });
  for(let i=hearts.length-1;i>=0;i--){
    const heart=hearts[i]; heart.y-=heart.speed; heart.angle+=heart.rotationSpeed;
    const s=heart.size;
    hCtx.save();
    hCtx.translate(heart.x,heart.y);
    hCtx.rotate(Math.sin(heart.angle)/10);
    hCtx.fillStyle='#ff69b4';
    hCtx.font=`${s}px Arial`;
    hCtx.fillText('‚ù§Ô∏è You',-s/2,s/2);
    hCtx.restore();
    if(heart.y+s<0) hearts.splice(i,1);
  }
  requestAnimationFrame(animateHearts);
}
spawnSparkles(); animateHearts();

// ---------- Photo overlay messages ----------
document.querySelectorAll('.photo-card').forEach(card=>{
  const messageBox = card.querySelector('.message-box p');
  messageBox.textContent = card.dataset.message;
  card.addEventListener('click',()=>{ card.classList.toggle('show-message'); if(card.classList.contains('show-message')) spawnHearts(15); });
});

// ---------- Page switch ----------
const page1=document.getElementById('page1');
const page2=document.getElementById('page2');
const goNext = document.getElementById('goNext');
goNext.addEventListener('click',()=>{

  setTimeout(()=>{
    page1.style.display='none';
    page2.classList.remove('hidden');
    startPage2Effects();
  },800);
});

// ---------- Page 2 Confetti & Floating Hearts ----------
const confettiCanvas = document.getElementById('confetti');
const confCtx = confettiCanvas.getContext('2d');
const heartsCanvas2 = document.getElementById('floatingHeartsCanvas');
const heartsCtx2 = heartsCanvas2.getContext('2d');
let W=window.innerWidth,H=window.innerHeight;
confettiCanvas.width=heartsCanvas2.width=W;
confettiCanvas.height=heartsCanvas2.height=H;

window.addEventListener('resize', ()=>{
  W=confettiCanvas.width=heartsCanvas2.width=window.innerWidth;
  H=confettiCanvas.height=heartsCanvas2.height=window.innerHeight;
});

let confetti=[], floatingHearts2=[];
const palette=['#ff3d00','#ffd600','#00e676','#2979ff','#d500f9','#ff80ab'];

class Confetti {
  constructor(x,y){ this.x=x; this.y=y; this.w=Math.random()*6+6; this.h=Math.random()*8+8; this.vx=Math.random()*4-2; this.vy=Math.random()*2+1; this.color=palette[Math.floor(Math.random()*palette.length)]; this.life=0; this.ttl=Math.random()*80+70; }
  update(){ this.x+=this.vx; this.y+=this.vy; this.vy+=0.03; this.life++; }
  draw(ctx){ ctx.fillStyle=this.color; ctx.fillRect(this.x,this.y,this.w,this.h); }
}

class FloatingHeart2 {
  constructor(){ this.size=Math.random()*20+25; this.x=Math.random()*W; this.y=H+this.size; this.speedY=Math.random()*0.8+0.3; this.speedX=Math.random()*0.8-0.4; this.opacity=Math.random()*0.5+0.5; this.wave=Math.random()*Math.PI*2; }
  update(){ this.y-=this.speedY; this.wave+=0.01; this.x+=Math.sin(this.wave)*0.5; if(this.y<-this.size){ this.y=H+this.size; this.x=Math.random()*W; } }
  draw(ctx){ ctx.save(); ctx.globalAlpha=this.opacity; ctx.font=`${this.size}px Arial`; ctx.fillStyle='#ff69b4'; ctx.fillText('PS',this.x,this.y); ctx.restore(); }
}

function spawnBurst(cx,cy,count=50){ for(let i=0;i<count;i++){ const c=new Confetti(cx,cy); const angle=Math.random()*Math.PI*2; const speed=Math.random()*4+2; c.vx=Math.cos(angle)*speed; c.vy=Math.sin(angle)*speed; confetti.push(c); } }

function animatePage2(){
  confCtx.clearRect(0,0,W,H);
  heartsCtx2.clearRect(0,0,W,H);
  floatingHearts2.forEach(h=>{ h.update(); h.draw(heartsCtx2); });
  for(let i=confetti.length-1;i>=0;i--){ const c=confetti[i]; c.update(); c.draw(confCtx); if(c.y>H+40||c.life>c.ttl) confetti.splice(i,1); }
  requestAnimationFrame(animatePage2);
}

document.getElementById('party').addEventListener('click',()=>spawnBurst(W/2,H/2,160));

function startPage2Effects(){
  for(let i=0;i<30;i++) floatingHearts2.push(new FloatingHeart2());
  setInterval(()=>spawnBurst(Math.random()*W*0.8+W*0.1, Math.random()*H*0.4+H*0.2, 20),1500);
  animatePage2();
}
